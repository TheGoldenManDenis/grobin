DEFPROCplayermovement
 REM player controls

 CASE ControlMode% OF
 WHEN Control%,Record%
  control% = (INKEY(-68)AND8) OR (INKEY(-99)AND4) OR (INKEY(-98)AND2) OR (INKEY(-67)AND1)
  IF ControlMode%=Record% THEN
   BPUT#InputFp%,control%
  ENDIF  
 WHEN Replay%
  control%=BGET#InputFp%
 ENDCASE

 L%=(control%AND2)<>0: R%=(control%AND1)<>0
 IF L%: pxm -= 0.5: IF pxm<=0 phorizflip%=TRUE :REM: move left
 IF R%: pxm += 0.5: IF pxm>=0 phorizflip%=FALSE :REM: move right
 IF control%AND4 IF ponsurface%: PROCdetach: pxm+= COS(pangle+mpi2)*25: pym+= SIN(pangle+mpi2)*25 :REM: jump
 REM --- debug ---
 IF control%AND8: pxm+=5*(phorizflip%-(phorizflip%=FALSE)) :REM: F: move fast
 REM IF INKEY(-51) IF ponsurface%: PROCdetach           :REM: D: detach
 WHILE INKEY(-71)ANDINKEY(-102):WAIT 1:ENDWHILE         :REM: hold K to freeze
 WHILE INKEY(-71)ANDNOTINKEY(-102):WAIT 1:ENDWHILE      :REM: press M to advance frame while frozen

 ox=px
 oy=py

 IF ponsurface% THEN
  pangle=FNplatangle(pcp%,px,py)
  pcl%=layer%(pcp%)
  REM apply slope gravity
  pxm-=SINang(pcp%)*0.7
  REM allow player to stop moving
  IF pangle>-PI*0.15 AND pangle<PI*0.15 THEN
   IF NOTL% IF NOTR% IF ABSpxm<6: pxm=pxm*0.995: IF pangle=0 IF ABSpxm<0.6: pxm=0
  ENDIF
  REM move player along ground 
  REM px+=COSang(pcp%)*pxm: py+=SINang(pcp%)*pxm
  REM momentum decay / drag
  pxm=pxm*0.99
  REM: check if player has moved off of current platform
  _d = ABSpxm
  WHILE _d>0 AND ponsurface%
   ox=px:oy=py
   IF pxm>=0 THEN
    cpl = FNgetdist(px,py,xb%(pcp%),yb%(pcp%))
    IF _d>=cpl THEN
     _d-=cpl
     IF lp%(pcp%): pcp%+=1:px=xa%(pcp%):py=ya%(pcp%):pangle=ang(pcp%) ELSE px=xb%(pcp%):py=yb%(pcp%):pangle=ang(pcp%): PROCdetach:PROCwallcollision(PI):PROCwallcollision(0): px+=pxm: py+=pym: GOTO 174
    ELSE
     px+=COSang(pcp%)*_d*SGNpxm: py+=SINang(pcp%)*_d*SGNpxm: _d=0
    ENDIF
   ELSE
    cpl = FNgetdist(px,py,xa%(pcp%),ya%(pcp%))
    IF _d>=cpl THEN
     _d-=cpl
     IF lm%(pcp%): pcp%-=1:px=xb%(pcp%):py=yb%(pcp%):pangle=ang(pcp%) ELSE px=xa%(pcp%):py=ya%(pcp%):pangle=ang(pcp%): PROCdetach:PROCwallcollision(PI):PROCwallcollision(0): px+=pxm: py+=pym: GOTO 174
    ELSE
     px+=COSang(pcp%)*_d*SGNpxm: py+=SINang(pcp%)*_d*SGNpxm: _d=0
    ENDIF
   ENDIF
   IF ponsurface% pangle=FNplatangle(pcp%,px,py)
   PROCwallcollision(PI):PROCwallcollision(0)
  ENDWHILE
  IF ponsurface% IF NOT FNboxcheck(xa%(pcp%),ya%(pcp%),xb%(pcp%),yb%(pcp%),px,py): PROCdetach
 ELSE
  pym -= 1     :REM: apply gravity
  pxm=pxm*0.9905 :REM: momentum decay / drag
  px+=pxm
  py+=pym
  REM check for floor collisions
  174 pc%=FNplatformcollision(pcl%,ox,oy,px,py)
  IF pc% THEN 
   IF ABSang(pc%)=mpi2: pxm=0:px=sx-SGNang(pc%): GOTO174
   pcp%=pc%
   px=sx
   py=sy
   ponsurface%=TRUE
   pangle=ang(pc%)
   IF pangle<PI*0.5 AND pangle>-(PI*0.5) THEN
    IF pym<0 THEN pxm += SINpangle*pym
   ELSE
    IF pym>0 THEN pxm -= SINpangle*pym
   ENDIF
   IF pangle>PI*0.5 OR pangle<-(PI*0.5) THEN pxm=-pxm
  ENDIF
  REM check for wall collisions
  PROCwallcollision(PI)
  PROCwallcollision(0)

  
 ENDIF
ENDPROC
DEFPROCdetach
 LOCAL x,y
 x=px:y=py
 ponsurface%=FALSE
 pym = SINpangle*pxm: pxm = COSpangle*pxm
 px += COS(pangle+mpi2)*f2sd
 py += SIN(pangle+mpi2)*f2sd-f2sd
 IF FNplatformcollision(pcl%,px,py+f2sd,px,py): py=sy
 cpx%-=px-x
 cpy%-=py-y
ENDPROC
DEFFNplatangle(p%,x,y)
 LOCAL a,_a,_d,pd
 a=ang(p%)
 IF lm%(p%) IF a<>0 IF ABSa <> PI IF ABSa<>mpi2 THEN _a = ang(p%-1): _d = FNgetdist(xa%(p%),ya%(p%),x,y): pd = FNgetdist(xa%(p%),ya%(p%),xb%(p%),yb%(p%)): = _a + FNwrap(a - _a) * (_d / pd) ELSE =a
DEFFNgetdist(x1,y1,x2,y2)
 x1-=x2:x1=x1*x1:y1-=y2:y1=y1*y1: =SQR(x1+y1)
 LOCAL dx,dy
 dx=x1-x2: dx=dx*dx
 dy=y1-y2: dy=dy*dy
=SQR(dx+dy)
DEFFNwrap(a)
 IFABSa>PI: =a-PI*2*SGNa
=a

REM ======= determine what side of a line a point px,py is on =========================

DEF FNlinecheck(x1,y1,x2,y2, px,py)
 IF x1=x2 THEN =px>x1 EOR y2<y1
 IF y1=y2 THEN =py<y1 EOR x2<x1
=py<y1+(y2-y1)/(x2-x1)*(px-x1) EOR (x2<x1)
 LOCAL slope,threshold
 slope=(y2-y1)/(x2-x1)
 threshold=y1+slope*(px-x1)
=py<threshold EOR (x2<x1)

DEF FNplatformcollision(l%,px1,py1,px2,py2)
 LOCAL i%,j%, x,y
 FOR i%=1 TO numplats%
  IF layer%(i%)=l% OR l%=-1 THEN
   IF FN_c IF NOT FNlinecheck(xa%(i%),ya%(i%),xb%(i%),yb%(i%), px1,py1) IF FNlinecheck(xa%(i%),ya%(i%),xb%(i%),yb%(i%), px2,py2):T%=TRUE ELSET%=FALSE
   IFT%THEN
    sx=px1: sy=py1
    ex=px2: ey=py2
    FOR j%=0 TO 200
     x=sx*0.5+ex*0.5
     y=sy*0.5+ey*0.5
   
     IF FNlinecheck(xa%(i%),ya%(i%),xb%(i%),yb%(i%), x,y) THEN 
      ex=x:ey=y
     ELSE 
      sx=x:sy=y
     ENDIF
    
    NEXT
    IF FNboxcheck(xa%(i%),ya%(i%),xb%(i%),yb%(i%), x,y) THEN =i%
   ENDIF :REM: endif linecheck
  ENDIF :REM: endif layer%(i%)=l%
 NEXT :REM: i%
=FALSE

DEF FNboxcheck(x1,y1,x2,y2, px,py)
 IFx1>x2 SWAP x1,x2
 IFy1>y2 SWAP y1,y2
 x1-=1:x2+=1
 y1-=1:y2+=1
= px>=x1 AND px<=x2 AND py>=y1 AND py<=y2

DEF PROCwallcollision(a)
 IFponsurface%THEN
  _a=pangle+mpi2:co=COS_a*f2sd:si=SIN_a*f2sd
  x1 = ox + co
  y1 = oy + si
  x2 = px + co +COSa*s2sd
  y2 = py + si +SINa*s2sd
 ELSE
  x1 = ox      
  y1 = oy+f2sd 
  x2 = px      +COSa*s2sd
  y2 = py+f2sd +SINa*s2sd
 ENDIF
 pc%=FNplatformcollision(pcl%,x1,y1,x2,y2)

 REM: debug
 IF0THEN
  GCOL 255,0,0:CIRCLEFILL x1-vx%,y1-vy%,10
  CIRCLE ox-vx%,oy-vy%,15
  GCOL 0,255,0:CIRCLEFILLx2-vx%,y2-vy%,10
  CIRCLE px-vx%,py-vy%,15
  GCOL 255,128,0: RECTANGLE ox-vx%,oy-vy%,px-ox,py-oy
  
  OSCLI"refresh":WAIT 1
 ENDIF

 IF pc% THEN
  IF NOT FNlinecheck(xa%(pc%),ya%(pc%),xb%(pc%),yb%(pc%),ox,oy) OR ponsurface% THEN
   px += sx-x2
   py += sy-y2
   cpx% -= sx-x2
   cpy% -= sy-y2
   IF ponsurface%: pxm=0 ELSE pxm=pxm*0.6
   d_=0
   REM OSCLI"beep -f 100 -l 10":CLS:PROCdrawplats:CIRCLEFILLsx-vx%,sy-vy%,15:OSCLI"refresh":END
  ENDIF
 ENDIF
ENDPROC

DEF FNfmod( a, b ): b=ABSb*SGNa: =a-b*INT(a/b)
DEF FNtw(i)
 i+=PI*0.5
 i=FNfmod(i,PI*2)
 IF i< PI THEN
  = -4 * (i/(PI*2) ) +1
 ELSE
  = 4 * ((i-PI)/(PI*2) ) -1
 ENDIF

DEFPROCdrawpic(p%, x,y,s,a, px%,py%)
 LOCAL w%,h%,i%,j%,c%, coUP,siUP, coRI,siRI, X,Y, S,pp%, ox%,oy%
 ox%=x
 oy%=y
 S = s*1.41421356*4
 w%=!p%: p%+=4
 h%=!p%: p%+=4
 coUP = s*COS(a+PI*0.5)
 siUP = s*SIN(a+PI*0.5)
 coRI = s*COSa
 siRI = s*SINa
 x=x-coRI*px%-coUP*py%
 y=y-siRI*px%-siUP*py%
 pp%=p%
 FOR j%=0 TO h%-1 STEP4
  X=x+coUP*j%
  Y=y+siUP*j%
  p%=pp%+2*w%*j%
  FOR i%=0 TO w%-1 STEP4
   c%=!p%AND65535
   IF c%: GCOL c%>>7AND248,c%>>2AND248,c%<<3AND248: RECTANGLEFILL X,Y,S
   X+=coRI*4
   Y+=siRI*4
   p%+=8
  NEXT
 NEXT
ENDPROC
 GCOL0
 CIRCLE ox%+4,oy%-4,10
 CIRCLE ox%+4,oy%-4,2
 GCOL127
 CIRCLE ox%,oy%,10
 CIRCLE ox%,oy%,2
ENDPROC

   ?(P%+1)=C%>>8
   P%+=2
  NEXT
 NEXT
=pic%
